泛化行为内核模板



```typescript

// 非选择类组件的确定性内核模板

interface BehaviorKernel<State, Action, Output> {

  // 1. 初始状态

  initialState: State;

  // 2. 确定性 reducer（相同 action + 相同 state → 相同新 state）

  reducer: (state: State, action: Action) => State;

  // 3. 铁律保障输出

  getOutput: (state: State) => Output;

  // 4. 交互处理器（内化 useStableHandler）

  createHandlers: (state: State, dispatch: (action: Action) => void) => Record<string, Function>;

}



// 示例：拖拽排序内核

const DragSortKernel: BehaviorKernel<Item[], DragAction, Item[]> = {

  initialState: [],

  reducer: (state, action) => {

    if (action.type === 'REORDER') {

      const newState = [...state];

      const [removed] = newState.splice(action.from, 1);

      newState.splice(action.to, 0, removed);

      return newState;

    }

    return state;

  },

  getOutput: (state) => state,

  createHandlers: (state, dispatch) => ({

    onDragEnd: (from: number, to: number) => {

      if (from === to) return;

      dispatch({ type: 'REORDER', from, to });

    },

  }),

};

```
