function useThrottledValue<T>(value: T, delay: number = 300): T {

  const [throttledValue, setThrottledValue] = useState(value);

  const lastRun = useRef(Date.now());

  const pending = useRef<NodeJS.Timeout | null>(null);



  useEffect(() => {

    const now = Date.now();

    if (now - lastRun.current >= delay) {

      lastRun.current = now;

      setThrottledValue(value);

    } else if (pending.current === null) {

      pending.current = setTimeout(() => {

        setThrottledValue(value);

        lastRun.current = Date.now();

        pending.current = null;

      }, delay - (now - lastRun.current));

    }

    return () => {

      if (pending.current) {

        clearTimeout(pending.current);

        pending.current = null;

      }

    };

  }, [value, delay]);



  return throttledValue;

}

function useVirtualScroll<T>(

  items: T[],

  itemHeight: number = 36,

  containerHeight: number = 400,

  threshold: number = 100

) {

  const shouldVirtualize = items.length > threshold;



  if (!shouldVirtualize) {

    return {

      visibleItems: items,

      startIndex: 0,

      totalHeight: items.length * itemHeight,

      scrollToIndex: () => {},

      isVirtualized: false,

    };

  }



  const [scrollTop, setScrollTop] = useState(0);

  const startIndex = Math.floor(scrollTop / itemHeight);

  const endIndex = Math.min(

    startIndex + Math.ceil(containerHeight / itemHeight) + 2,

    items.length

  );



  const visibleItems = useMemo(

    () => items.slice(startIndex, endIndex),

    [items, startIndex, endIndex]

  );



  return {

    visibleItems,

    startIndex,

    totalHeight: items.length * itemHeight,

    scrollToIndex: useCallback((index: number) => {

      setScrollTop(Math.max(0, Math.min(index * itemHeight, totalHeight - containerHeight)));

    }, [itemHeight, totalHeight, containerHeight]),

    isVirtualized: true,

    onScroll: (e: React.UIEvent<HTMLElement>) => {

      setScrollTop(e.currentTarget.scrollTop);

    },

    containerStyle: { height: containerHeight, overflowY: 'auto' },

    spacerStyle: { height: items.length * itemHeight, position: 'relative' },

    itemStyle: (index: number) => ({

      position: 'absolute',

      top: index * itemHeight,

      height: itemHeight,

      width: '100%',

    }),

  };

}

interface SelectionKernelConfig<T> {

  items?: TreeNode<T>[];

  value?: T | T[] | null;

  defaultValue?: T | T[] | null;

  mode: 'single' | 'multiple';

  onSelect?: (val: T | T[]) => void;

  onBeforeSelect?: (newVal: T | T[], oldVal: T | T[] | null) => boolean | Promise<boolean>;

  onTransformSelect?: (rawVal: T | T[], oldVal: T | T[] | null) => T | T[];

  onAfterSelect?: (newVal: T | T[], oldVal: T | T[] | null) => void;

  selectDebounce?: number;

  enableFilter?: boolean;

  cascade?: boolean;

}



function useSelection<T>(config: SelectionKernelConfig<T>) {

  // L3: 使用 P3 稳定事件处理器

  const stableOnSelect = useStableHandler(config.onSelect || (() => {}));

  const stableOnBeforeSelect = useStableHandler(config.onBeforeSelect);

  const stableOnTransformSelect = useStableHandler(config.onTransformSelect);

  const stableOnAfterSelect = useStableHandler(config.onAfterSelect);



  const isControlled = config.value !== undefined;

  const [internalValue, setInternalValue] = useState<T | T[] | null>(

    config.defaultValue ?? null

  );

  const selectedValue = isControlled ? config.value : internalValue;



  const isEqual = useMemo(

    () => config.isEqual || ((a: T, b: T) => a === b),

    [config.isEqual]

  );



  const isSelected = useCallback(

    (itemValue: T): boolean => {

      if (selectedValue === null || selectedValue === undefined) return false;

      if (config.mode === 'multiple') {

        return Array.isArray(selectedValue) && selectedValue.some(v => isEqual(v, itemValue));

      }

      return isEqual(selectedValue as T, itemValue);

    },

    [selectedValue, isEqual, config.mode]

  );



  // L3: 完整异步流水线

  const setSelectedValue = useStableHandler(async (newVal: T | T[]) => {

    const oldVal = selectedValue;



    if (stableOnBeforeSelect) {

      const canProceed = await stableOnBeforeSelect(newVal, oldVal);

      if (canProceed === false) return;

    }



    let finalVal = newVal;

    if (stableOnTransformSelect) {

      finalVal = stableOnTransformSelect(newVal, oldVal);

    }



    const isSame = config.mode === 'multiple'

      ? Array.isArray(finalVal) && Array.isArray(oldVal) &&

        finalVal.length === oldVal.length &&

        finalVal.every((v, i) => isEqual(v, oldVal[i]))

      : isEqual(finalVal as T, oldVal as T);

    if (isSame) return;



    if (isControlled) {

      stableOnSelect(finalVal);

    } else {

      setInternalValue(finalVal);

      stableOnSelect(finalVal);

    }



    queueMicrotask(() => {

      stableOnAfterSelect?.(finalVal, oldVal);

    });

  });



  const getItemProps = useCallback(

    (item: TreeNode<T>) => ({

      role: config.mode === 'multiple' ? 'checkbox' : 'radio',

      'aria-checked': isSelected(item.value),

      'aria-disabled': item.disabled || false,

      tabIndex: isSelected(item.value) ? 0 : -1,

      onClick: () => {

        if (item.disabled) return;

        if (config.mode === 'multiple') {

          toggleSelection(item.value);

        } else {

          setSelectedValue(item.value);

        }

      },

    }),

    [isSelected, setSelectedValue, toggleSelection, config.mode]

  );



  const toggleSelection = useStableHandler((itemValue: T) => {

    if (config.mode !== 'multiple') return;

    const current = Array.isArray(selectedValue) ? selectedValue : [];

    const exists = current.some(v => isEqual(v, itemValue));

    const newVal = exists

      ? current.filter(v => !isEqual(v, itemValue))

      : [...current, itemValue];

    setSelectedValue(newVal);

  });



  return {

    selectedValue,

    setSelectedValue,

    isSelected,

    toggleSelection,

    getItemProps,

    loading: false,

    mode: config.mode,

  };

}
