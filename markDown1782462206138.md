第零部分：元铁律（Meta-Law）—— 不可变基石



铁律声明（强制执行）：

UI = f(Props)。对于相同的 Props 输入，组件必须永远返回完全相同的虚拟 DOM 输出。

此铁律适用于 L1/L2/L3 所有层级、M（元内核）的所有 Hook、P（性能层）的所有优化函数和 S（实例）的所有渲染函数。



禁止行为（全局）：



· 内部使用 Math.random()、Date.now()、useId() 生成不稳定 ID

· useEffect 监听全局尺寸（window.innerWidth/resize）

· 依赖任何外部可变状态（如 navigator.language）

· 列表渲染使用 index 作为 key

· 使用 JSON.stringify 作为 arePropsEqual 的深度比较（大数据量下性能灾难）



铁律的 4 个强制检查点：



检查点 实现方式 违反后果

Props 不变，输出不变 React.memo + createStableComparator（指定深度比较字段） 无效重渲染

无隐式副作用 useEffect 依赖数组完整，无 window/document 依赖 输入相同但输出不同

稳定标识符 列表 key 使用 item.value 或稳定哈希 组件卸载/重挂载导致状态丢失

性能稳定 比较器耗时 < 1ms（1000+ 项），事件函数引用稳定 UI 卡顿或子组件失效

P1：智能比较器工厂（createStableComparator）



```typescript

/**

 * 智能比较器工厂

 * @param deepKeys - 需要深度比较的字段列表（如 ['items', 'value']）

 * @param customComparators - 自定义比较逻辑

 */

function createStableComparator<T extends Record<string, any>>(

  deepKeys: (keyof T)[] = ['items', 'value', 'options', 'data'],

  customComparators?: Partial<Record<keyof T, (a: any, b: any) => boolean>>

) {

  return (prev: T, next: T): boolean => {

    if (prev === next) return true;



    const allKeys = new Set([...Object.keys(prev), ...Object.keys(next)]);

    for (const key of allKeys) {

      const prevVal = (prev as any)[key];

      const nextVal = (next as any)[key];



      if (key === 'children' || key === 'renderSkeleton' || key === 'renderError') continue;

      if (typeof prevVal === 'function') continue;

      if (prevVal === nextVal) continue;



      if (deepKeys.includes(key as keyof T)) {

        const customComparator = customComparators?.[key as keyof T];

        if (customComparator) {

          if (!customComparator(prevVal, nextVal)) return false;

        } else {

          if (!isEqualDeep(prevVal, nextVal)) return false;

        }

      } else {

        return false;

      }

    }

    return true;

  };

}



function isEqualDeep(a: any, b: any): boolean {

  if (a === b) return true;

  if (Array.isArray(a) && Array.isArray(b)) {

    if (a.length !== b.length) return false;

    const limit = Math.min(a.length, b.length, 100);

    for (let i = 0; i < limit; i++) {

      if (!isEqualDeep(a[i], b[i])) return false;

    }

    return true;

  }

  if (typeof a === 'object' && a !== null && typeof b === 'object' && b !== null) {

    const keysA = Object.keys(a);

    const keysB = Object.keys(b);

    if (keysA.length !== keysB.length) return false;

    for (const key of keysA) {

      if (a[key] !== b[key]) return false;

    }

    return true;

  }

  return false;

}

```

function createScopedContext<TSelection, TConfig>() {

  const SelectionContext = createContext<TSelection | null>(null);

  const ConfigContext = createContext<TConfig | null>(null);



  function Provider({ children, selection, config }: {

    children: React.ReactNode;

    selection: TSelection;

    config: TConfig;

  }) {

    const stableConfig = useMemo(() => config, [config.theme, config.dir, config.size]);

    return (

      <ConfigContext.Provider value={stableConfig}>

        <SelectionContext.Provider value={selection}>

          {children}

        </SelectionContext.Provider>

      </ConfigContext.Provider>

    );

  }



  return { Provider, SelectionContext, ConfigContext };

}

function useStableHandler<T extends (...args: any[]) => any>(handler: T): T {

  const ref = useRef(handler);

  ref.current = handler;

  return useCallback((...args: Parameters<T>) => ref.current(...args), []) as T;

}



function useStableAsyncHandler<T extends (...args: any[]) => Promise<any>>(

  handler: T,

  onPending?: (isPending: boolean) => void

): { execute: T; isPending: boolean } {

  const [isPending, setIsPending] = useState(false);

  const stableHandler = useStableHandler(handler);



  const execute = useCallback(

    async (...args: Parameters<T>) => {

      setIsPending(true);

      onPending?.(true);

      try {

        return await stableHandler(...args);

      } finally {

        setIsPending(false);

        onPending?.(false);

      }

    },

    [stableHandler, onPending]

  ) as T;



  return { execute, isPending };

}
