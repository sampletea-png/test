interface SpaceAdaptationConfig {

  responsive?: boolean;

  breakpoint?: string;

  collapseBehavior?: 'vertical' | 'wrap' | 'scroll';

  labelOverflow?: 'ellipsis' | 'wrap' | 'shrink';

  maxLabelWidth?: number;

  compact?: boolean;

}



function useSpaceAdaptation(config: SpaceAdaptationConfig) {

  // ⚠️ 铁律：纯 CSS 媒体查询，禁止 useEffect 监听 resize



  const rootClassName = useMemo(() => {

    const classes: string[] = ['radio-group'];

    if (config.responsive) classes.push('radio-group-responsive');

    if (config.compact) classes.push('radio-group-compact');

    return classes.join(' ');

  }, [config.responsive, config.compact]);



  const itemStyle = useMemo<React.CSSProperties>(() => {

    if (config.compact) return { padding: '2px 4px', fontSize: '10px' };

    return {};

  }, [config.compact]);



  const labelStyle = useMemo<React.CSSProperties>(() => {

    if (config.labelOverflow === 'ellipsis') {

      return {

        whiteSpace: 'nowrap',

        overflow: 'hidden',

        textOverflow: 'ellipsis',

        ...(config.maxLabelWidth ? { maxWidth: `${config.maxLabelWidth}px` } : {}),

      };

    }

    if (config.labelOverflow === 'shrink') {

      return { fontSize: 'clamp(10px, 2vw, 14px)' };

    }

    return {};

  }, [config.labelOverflow, config.maxLabelWidth]);



  const mediaQueryCSS = useMemo(() => {

    if (!config.responsive) return '';

    const collapseMap = {

      vertical: 'flex-direction: column;',

      wrap: 'flex-wrap: wrap;',

      scroll: 'overflow-x: auto; flex-wrap: nowrap;',

    };

    return `

      @media (max-width: ${config.breakpoint || '600px'}) {

        .radio-group-responsive { ${collapseMap[config.collapseBehavior || 'wrap']} }

      }

    `;

  }, [config.responsive, config.breakpoint, config.collapseBehavior]);



  useEffect(() => {

    if (!mediaQueryCSS) return;

    const styleEl = document.createElement('style');

    styleEl.textContent = mediaQueryCSS;

    document.head.appendChild(styleEl);

    return () => styleEl.remove();

  }, [mediaQueryCSS]);



  return { rootClassName, itemStyle, labelStyle };

}

interface DesignTokens {

  primaryColor: string;

  primaryHoverColor: string;

  primaryActiveColor: string;

  primaryBorderColor: string;

  textColor: string;

  disabledColor: string;

  errorColor: string;

  fontSizeBase: number;

  spacingBase: number;

  borderRadius: number;

  transitionDuration: string;

  transitionTiming: string;

}



const defaultTheme: DesignTokens = {

  primaryColor: '#1677FF',

  primaryHoverColor: '#E6F4FF',

  primaryActiveColor: '#0958D9',

  primaryBorderColor: '#D9D9D9',

  textColor: '#333333',

  disabledColor: '#BFBFBF',

  errorColor: '#FF4D4F',

  fontSizeBase: 14,

  spacingBase: 8,

  borderRadius: 4,

  transitionDuration: '0.2s',

  transitionTiming: 'ease-in-out',

};



function buildUniversalStyles(params: {

  theme?: Partial<DesignTokens>;

  size?: 'small' | 'medium' | 'large';

  dir?: 'ltr' | 'rtl';

  disabled?: boolean;

  invalid?: boolean;

  gap?: number;

  bordered?: boolean;

  rounded?: 'none' | 'small' | 'medium' | 'full';

  shadow?: boolean;

  fullWidth?: boolean;

}) {

  const theme = { ...defaultTheme, ...params.theme };

  const isRTL = params.dir === 'rtl';



  const sizeMap = {

    small: { fontSize: theme.fontSizeBase - 2, padding: theme.spacingBase / 2 },

    medium: { fontSize: theme.fontSizeBase, padding: theme.spacingBase },

    large: { fontSize: theme.fontSizeBase + 2, padding: theme.spacingBase * 1.5 },

  };

  const size = sizeMap[params.size || 'medium'];



  const roundedMap = { none: 0, small: theme.borderRadius / 2, medium: theme.borderRadius, full: 9999 };

  const borderRadius = roundedMap[params.rounded || 'medium'];



  const paddingStart = isRTL ? 'paddingRight' : 'paddingLeft';

  const paddingEnd = isRTL ? 'paddingLeft' : 'paddingRight';

  const marginStart = isRTL ? 'marginRight' : 'marginLeft';



  return {

    root: {

      display: 'flex',

      flexDirection: 'row',

      flexWrap: 'wrap',

      gap: theme.spacingBase,

      ...(params.fullWidth ? { width: '100%' } : {}),

      ...(params.invalid ? { borderColor: theme.errorColor, backgroundColor: `${theme.errorColor}10` } : {}),

    },

    item: {

      display: 'flex',

      alignItems: 'center',

      cursor: params.disabled ? 'not-allowed' : 'pointer',

      opacity: params.disabled ? 0.6 : 1,

      fontSize: `${size.fontSize}px`,

      padding: `${size.padding}px`,

      borderRadius: `${borderRadius}px`,

      [paddingStart]: theme.spacingBase,

      [paddingEnd]: theme.spacingBase,

      ...(params.bordered ? { border: `1px solid ${theme.primaryBorderColor}` } : {}),

      ...(params.shadow ? { boxShadow: '0 2px 8px rgba(0,0,0,0.08)' } : {}),

    },

    checked: {

      backgroundColor: theme.primaryHoverColor,

      borderColor: theme.primaryColor,

      color: theme.primaryColor,

    },

    input: {

      [marginStart]: 0,

      [marginEnd]: params.gap ?? theme.spacingBase,

      accentColor: theme.primaryColor,

      ...(params.invalid ? { accentColor: theme.errorColor } : {}),

    },

    label: { color: params.disabled ? theme.disabledColor : theme.textColor },

    error: { color: theme.errorColor, fontSize: `${theme.fontSizeBase - 2}px`, marginTop: '2px', display: 'block' },

  };

}

interface UIAdapter<T = any> {

  createElement: (tag: string, props: any, ...children: any[]) => T;

  useState: <S>(initial: S) => [S, (s: S) => void];

  useEffect: (fn: () => void | (() => void), deps: any[]) => void;

  useMemo: <U>(fn: () => U, deps: any[]) => U;

  useCallback: <F extends Function>(fn: F, deps: any[]) => F;

  useRef: <U>(initial: U) => { current: U };

  forwardRef: (fn: any) => any;

  useImperativeHandle: (ref: any, fn: () => any, deps: any[]) => void;

  memo: (fn: any, cmp?: any) => any;

  useContext: <C>(context: any) => C;

  createContext: <C>(defaultValue: C) => any;

}
