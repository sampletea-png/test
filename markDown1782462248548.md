function useTreeTraversal<T>(items: TreeNode<T>[]) {

  const flattenTree = useMemo(() => {

    const result: TreeNode<T>[] = [];

    const traverse = (node: TreeNode<T>, depth: number) => {

      result.push({ ...node, depth });

      if (node.children) {

        node.children.forEach(child => traverse(child, depth + 1));

      }

    };

    items.forEach(item => traverse(item, 0));

    return result;

  }, [items]);



  const findNode = useCallback((value: T): TreeNode<T> | null => {

    const search = (node: TreeNode<T>): TreeNode<T> | null => {

      if (node.value === value) return node;

      if (node.children) {

        for (const child of node.children) {

          const found = search(child);

          if (found) return found;

        }

      }

      return null;

    };

    for (const item of items) {

      const found = search(item);

      if (found) return found;

    }

    return null;

  }, [items]);



  const getAncestors = useCallback((value: T): TreeNode<T>[] => {

    const result: TreeNode<T>[] = [];

    const search = (node: TreeNode<T>, path: TreeNode<T>[]): boolean => {

      const currentPath = [...path, node];

      if (node.value === value) {

        result.push(...currentPath);

        return true;

      }

      if (node.children) {

        for (const child of node.children) {

          if (search(child, currentPath)) return true;

        }

      }

      return false;

    };

    for (const item of items) {

      if (search(item, [])) break;

    }

    return result;

  }, [items]);



  const getDescendants = useCallback((value: T): TreeNode<T>[] => {

    const node = findNode(value);

    if (!node || !node.children) return [];

    const result: TreeNode<T>[] = [];

    const traverse = (n: TreeNode<T>) => {

      result.push(n);

      if (n.children) n.children.forEach(traverse);

    };

    node.children.forEach(traverse);

    return result;

  }, [findNode]);



  return { flattenTree, findNode, getAncestors, getDescendants };

}

function useKeyboardNavigation<T>(props: {

  items: TreeNode<T>[];

  selectedValue: T | T[] | null;

  onSelect: (val: T) => void;

  dir: 'ltr' | 'rtl';

  expandable?: boolean;

}) {

  const [focusedIndex, setFocusedIndex] = useState<number>(-1);

  const flatItems = useTreeTraversal(props.items).flattenTree;

  const isRTL = props.dir === 'rtl';

  const nextKey = isRTL ? 'ArrowLeft' : 'ArrowRight';

  const prevKey = isRTL ? 'ArrowRight' : 'ArrowLeft';



  const getKeyDownHandler = useCallback(

    (e: React.KeyboardEvent) => {

      const currentIndex = focusedIndex >= 0 ? focusedIndex : 0;

      let newIndex = currentIndex;



      switch (e.key) {

        case 'ArrowDown': e.preventDefault(); newIndex = Math.min(currentIndex + 1, flatItems.length - 1); break;

        case 'ArrowUp': e.preventDefault(); newIndex = Math.max(currentIndex - 1, 0); break;

        case nextKey: e.preventDefault(); newIndex = Math.min(currentIndex + 1, flatItems.length - 1); break;

        case prevKey: e.preventDefault(); newIndex = Math.max(currentIndex - 1, 0); break;

        case 'Enter':

        case ' ':

          e.preventDefault();

          if (flatItems[newIndex]) props.onSelect(flatItems[newIndex].value);

          break;

        default: return;

      }



      if (newIndex !== currentIndex) setFocusedIndex(newIndex);

    },

    [focusedIndex, flatItems, props.onSelect, props.dir, nextKey, prevKey]

  );



  return { getKeyDownHandler, focusedIndex };

}

function useAsyncDataSource<T>(

  fetcher?: (keyword?: string, page?: number) => Promise<T[]>,

  options?: { debounce?: number; pageSize?: number }

) {

  const [items, setItems] = useState<T[]>([]);

  const [status, setStatus] = useState<'idle' | 'loading' | 'success' | 'error'>('idle');

  const [error, setError] = useState<Error | null>(null);

  const [keyword, setKeyword] = useState('');

  const [page, setPage] = useState(1);

  const [hasMore, setHasMore] = useState(true);

  const abortControllerRef = useRef<AbortController | null>(null);



  const fetchData = useCallback(async (resetPage: boolean = true) => {

    if (!fetcher) return;

    if (abortControllerRef.current) abortControllerRef.current.abort();

    const controller = new AbortController();

    abortControllerRef.current = controller;



    const currentPage = resetPage ? 1 : page;

    if (resetPage) { setPage(1); setItems([]); }



    setStatus('loading');

    setError(null);

    try {

      const result = await fetcher(keyword || undefined, currentPage);

      if (controller.signal.aborted) return;

      if (resetPage) setItems(result);

      else setItems(prev => [...prev, ...result]);

      setHasMore(result.length >= (options?.pageSize || 20));

      setStatus('success');

    } catch (err) {

      if (err instanceof Error && err.name === 'AbortError') return;

      setError(err instanceof Error ? err : new Error(String(err)));

      setStatus('error');

    }

  }, [fetcher, keyword, page, options?.pageSize]);



  const debouncedKeyword = useThrottledValue(keyword, options?.debounce || 300);

  useEffect(() => { if (fetcher) fetchData(true); }, [debouncedKeyword, fetcher]);



  return { items, status, error, retry: () => fetchData(true), loadMore: () => fetchData(false), hasMore, setKeyword };

}
